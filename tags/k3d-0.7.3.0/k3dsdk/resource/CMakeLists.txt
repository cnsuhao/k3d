PROJECT(k3dsdk-resource)

FILE(GLOB HEADERS *.h)
FILE(GLOB SOURCES *.cpp)

IF(CMAKE_HAS_SORT)
	LIST(SORT HEADERS)
	LIST(SORT SOURCES)
ENDIF(CMAKE_HAS_SORT)

ADD_LIBRARY(k3dsdk-resource SHARED ${HEADERS} ${SOURCES})
K3D_GENERATE_DEF_FILE(k3dsdk-resource)

TARGET_LINK_LIBRARIES(k3dsdk-resource k3dsdk)

INSTALL(TARGETS k3dsdk-resource
	RUNTIME DESTINATION bin
	LIBRARY DESTINATION lib
	ARCHIVE DESTINATION lib)
INSTALL(FILES ${HEADERS} DESTINATION include/k3d/k3dsdk/resource)
