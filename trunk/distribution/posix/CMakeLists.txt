# Generate self-extracting tarballs
ADD_CUSTOM_TARGET(package-binary-stgz
	${CMAKE_CPACK_COMMAND}
	-G STGZ
	-D "CPACK_CMAKE_GENERATOR=${CMAKE_GENERATOR}"
	-D "\"CPACK_INSTALL_CMAKE_PROJECTS=${K3D_CPACK_INSTALL_CMAKE_PROJECTS}\""
	-D "CPACK_PACKAGE_DESCRIPTION=${K3D_CPACK_PACKAGE_DESCRIPTION}"
	-D "CPACK_PACKAGE_FILE_NAME=${K3D_CPACK_PACKAGE_FILE_NAME}"
	-D "CPACK_PACKAGE_NAME=${K3D_CPACK_PACKAGE_NAME}"
	-D "CPACK_PACKAGE_VERSION=${K3D_CPACK_PACKAGE_VERSION}"
	-D "CPACK_RESOURCE_FILE_LICENSE=${K3D_CPACK_RESOURCE_FILE_LICENSE}"
	)
